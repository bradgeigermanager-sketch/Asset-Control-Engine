### Generator script to produce full datasets
Save this Python script as `generate_seeds.py`. Run `python generate_seeds.py` to create the full CSVs. It produces deterministic, varied data and injects the conflict and anomaly cases described above.

```python
# generate_seeds.py
import csv
import uuid
import random
from datetime import datetime, timedelta

OUT_ASSETS = "assets_seed.csv"
OUT_DISC = "discoveries_seed.csv"
OUT_ANOM = "anomalies_seed.csv"

def rand_mac(i):
    base = 0x001122334400 + i
    return ":".join(f"{(base >> (8*j)) & 0xff:02x}" for j in reversed(range(6)))

def iso(dt):
    return dt.strftime("%Y-%m-%dT%H:%M:%SZ")

def gen_assets(n=500):
    types = ["laptop","desktop","server","switch","printer","iot","vm"]
    man_models = [
        ("Dell","Latitude 7420"),("Apple","MacBook Pro"),("HP","ProLiant"),
        ("Cisco","Catalyst-9300"),("HP","ColorLaser"),("VendorX","SensorX"),
        ("VMware","UbuntuVM")
    ]
    rows = []
    now = datetime.utcnow()
    for i in range(1, n+1):
        aid = str(uuid.uuid4())
        tag = f"AT-{i:04d}"
        sn = f"SN-ABC-{1000 + i}"
        t, mm = random.choice(types), random.choice(man_models)
        hostname = f"user-{i}-host"
        mac = rand_mac(i)
        ip = f"10.0.{i%250}.{10 + (i%200)}"
        ownerId = f"u-{1000 + (i%200)}"
        ownerName = f"User {i}"
        costCenter = f"CC-{100 + (i%50)}"
        purchase = (now - timedelta(days=random.randint(100,1500))).date().isoformat()
        warranty = (now + timedelta(days=random.randint(90,900))).date().isoformat()
        lifecycle = "active"
        metadata = "{}"
        created = iso(now - timedelta(days=random.randint(1,30)))
        updated = iso(now - timedelta(days=random.randint(0,5)))
        rows.append({
            "id":aid,"assetTag":tag,"serialNumber":sn,"type":t,
            "manufacturer":mm[0],"model":mm[1],"hostname":hostname,
            "macAddresses":mac,"ipAddresses":ip,"ownerId":ownerId,"ownerName":ownerName,
            "costCenter":costCenter,"location":"HQ","purchaseDate":purchase,
            "warrantyEnd":warranty,"lifecycleState":lifecycle,"metadata":metadata,
            "createdAt":created,"updatedAt":updated
        })
    return rows

def gen_discoveries(assets, n_match=100, n_conflict=30, n_orphan=20):
    rows = []
    now = datetime.utcnow()
    # exact matches
    for i in range(n_match):
        a = assets[i]
        rows.append({
            "id":f"d-{i+1:04d}",
            "discoverySource": random.choice(["agent","mdm","network","cloud"]),
            "rawPayload":"{}",
            "serialNumber": a["serialNumber"],
            "hostname": a["hostname"],
            "macAddresses": a["macAddresses"],
            "ipAddresses": a["ipAddresses"],
            "os.name":"Windows",
            "os.version":"10",
            "lastSeen": iso(now - timedelta(minutes=random.randint(1,120))),
            "reconciliationStatus":"unmatched"
        })
    # conflicts: modify hostname or mac slightly
    for i in range(n_conflict):
        a = assets[n_match + i]
        mac = a["macAddresses"]
        # flip last nibble
        parts = mac.split(":")
        last = int(parts[-1],16) ^ 0x01
        parts[-1] = f"{last:02x}"
        rows.append({
            "id":f"d-{n_match + i +1:04d}",
            "discoverySource": random.choice(["network","agent"]),
            "rawPayload":"{}",
            "serialNumber": "" if random.random() < 0.6 else a["serialNumber"],
            "hostname": a["hostname"] + ("-CLONE" if random.random() < 0.5 else ""),
            "macAddresses": ":".join(parts),
            "ipAddresses": a["ipAddresses"],
            "os.name":"Unknown",
            "os.version":"",
            "lastSeen": iso(now - timedelta(minutes=random.randint(10,500))),
            "reconciliationStatus":"unmatched"
        })
    # orphans
    for i in range(n_orphan):
        idx = n_match + n_conflict + i
        rows.append({
            "id":f"d-{idx+1:04d}",
            "discoverySource": random.choice(["network","cloud"]),
            "rawPayload":"{}",
            "serialNumber": "",
            "hostname": f"unknown-{i+1}",
            "macAddresses": rand_mac(10000 + i),
            "ipAddresses": f"10.1.{i%250}.{20 + i%200}",
            "os.name":"Unknown",
            "os.version":"",
            "lastSeen": iso(now - timedelta(minutes=random.randint(5,1000))),
            "reconciliationStatus":"unmatched"
        })
    return rows

def gen_anomalies():
    rows = [
        {"id":"an-0001","type":"duplicate_mac","description":"Two canonical assets share same MAC 00:11:22:33:44:EE","discoveryId":"","notes":"seeded conflict"},
        {"id":"an-0002","type":"missing_serial","description":"Discovery payload missing serial and hostname","discoveryId":"d-0003","notes":"orphan"},
        {"id":"an-0003","type":"spoofed_mac","description":"MAC differs by one nibble from canonical","discoveryId":"d-0009","notes":"fuzzy match expected"},
        {"id":"an-0004","type":"malformed_payload","description":"Discovery rawPayload contains invalid JSON","discoveryId":"d-0012","notes":"should be rejected"},
        {"id":"an-0005","type":"stale_record","description":"Discovery lastSeen older than 90 days","discoveryId":"d-0200","notes":"should be archived"},
    ]
    return rows

def write_csv(path, rows, headers):
    with open(path, "w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=headers)
        w.writeheader()
        for r in rows:
            w.writerow(r)

if __name__ == "__main__":
    assets = gen_assets(500)
    discoveries = gen_discoveries(assets, n_match=100, n_conflict=30, n_orphan=20)
    anomalies = gen_anomalies()
    asset_headers = ["id","assetTag","serialNumber","type","manufacturer","model","hostname","macAddresses","ipAddresses","ownerId","ownerName","costCenter","location","purchaseDate","warrantyEnd","lifecycleState","metadata","createdAt","updatedAt"]
    disc_headers = ["id","discoverySource","rawPayload","serialNumber","hostname","macAddresses","ipAddresses","os.name","os.version","lastSeen","reconciliationStatus"]
    an_headers = ["id","type","description","discoveryId","notes"]
    write_csv(OUT_ASSETS, assets, asset_headers)
    write_csv(OUT_DISC, discoveries, disc_headers)
    write_csv(OUT_ANOM, anomalies, an_headers)
    print("Generated:", OUT_ASSETS, OUT_DISC, OUT_ANOM)
```

### Pytest reconciliation test suite
Save as `tests/test_reconciliation.py`. Configure `pytest` to run against your staging base URL via environment variable `API_BASE` and `API_TOKEN`. The suite seeds assets, posts discoveries, triggers reconciliation suggestions, runs reconciliation actions, and computes metrics. It fails the CI if coverage or false positive thresholds are not met.

```python
# tests/test_reconciliation.py
import os
import requests
import csv
import time
import json
import pytest

API_BASE = os.getenv("API_BASE", "https://staging.api.example.com")
TOKEN = os.getenv("API_TOKEN", "changeme")
HEADERS = {"Authorization": f"Bearer {TOKEN}", "Content-Type": "application/json"}

def post_asset(asset):
    url = f"{API_BASE}/api/assets"
    r = requests.post(url, headers=HEADERS, json=asset)
    r.raise_for_status()
    return r.json()

def post_discovery(d):
    url = f"{API_BASE}/api/discovery"
    r = requests.post(url, headers=HEADERS, json=d)
    r.raise_for_status()
    return r.json()

def reconcile(discoveryId, action, assetId=None):
    url = f"{API_BASE}/api/discovery/reconcile"
    payload = {"discoveryId": discoveryId, "action": action}
    if assetId:
        payload["assetId"] = assetId
    r = requests.post(url, headers=HEADERS, json=payload)
    r.raise_for_status()
    return r.json()

def load_csv(path):
    with open(path, newline='', encoding='utf-8') as f:
        return list(csv.DictReader(f))

@pytest.fixture(scope="session")
def seed_data(tmp_path_factory):
    assets = load_csv("assets_seed.csv")
    discoveries = load_csv("discoveries_seed.csv")
    # seed assets
    created = []
    for a in assets:
        payload = {
            "id": a["id"],
            "assetTag": a["assetTag"],
            "serialNumber": a["serialNumber"],
            "type": a["type"],
            "hostname": a["hostname"],
            "macAddresses": [a["macAddresses"]],
            "ipAddresses": [a["ipAddresses"]],
            "owner": {"id": a["ownerId"], "name": a["ownerName"]},
            "lifecycleState": a["lifecycleState"]
        }
        post_asset(payload)
        created.append(a["id"])
    # post discoveries
    posted = []
    for d in discoveries:
        payload = {
            "id": d["id"],
            "discoverySource": d["discoverySource"],
            "rawPayload": {},
            "serialNumber": d["serialNumber"],
            "hostname": d["hostname"],
            "macAddresses": [d["macAddresses"]] if d["macAddresses"] else [],
            "ipAddresses": [d["ipAddresses"]] if d["ipAddresses"] else [],
            "lastSeen": d["lastSeen"]
        }
        post_discovery(payload)
        posted.append(d["id"])
    return {"assets": created, "discoveries": posted}

def test_exact_serial_match(seed_data):
    # pick a discovery that has serial matching asset AT-0001
    disc_id = "d-0001"
    # attempt reconcile by linking to known asset id
    resp = reconcile(disc_id, "link", assetId="a1e8f9b2-0001")
    assert resp.get("success") is True

def test_coverage_and_false_positive_rate(seed_data):
    # fetch unmatched discoveries
    r = requests.get(f"{API_BASE}/api/discovery/unmatched", headers=HEADERS)
    r.raise_for_status()
    items = r.json().get("items", [])
    # trigger suggestions for each and auto-link exact serial matches
    matched = 0
    false_positives = 0
    expected_matches = 100  # per dataset design
    for d in items:
        # if serial present and matches seeded asset serial, link automatically
        serial = d.get("serialNumber")
        if serial:
            # find asset by serial
            q = requests.get(f"{API_BASE}/api/assets?q={serial}", headers=HEADERS)
            if q.status_code == 200 and q.json().get("items"):
                asset = q.json()["items"][0]
                rec = reconcile(d["id"], "link", assetId=asset["id"])
                if rec.get("success"):
                    matched += 1
                    # verify correctness by checking asset serial equals expected
                    if asset.get("serialNumber") != serial:
                        false_positives += 1
    coverage = matched / expected_matches if expected_matches else 0
    false_positive_rate = false_positives / matched if matched else 0
    # thresholds
    assert coverage >= 0.95, f"Coverage too low: {coverage:.2%}"
    assert false_positive_rate < 0.02, f"False positive rate too high: {false_positive_rate:.2%}"
```

**Notes**
- The test suite assumes the API supports `POST /api/discovery` for direct posting; if your implementation uses a different endpoint, adjust the `post_discovery` function.
- The test uses simple serial lookup via `GET /api/assets?q=serial`. If your API uses a different filter, adapt accordingly.
- Add more tests for conflict handling, orphan create flows, malformed payload handling, and connector error simulation.

### CI pipeline template
Use this GitHub Actions workflow as `/.github/workflows/reconciliation-ci.yml`. It seeds data, runs tests, and fails on thresholds.

```yaml
name: Reconciliation Tests

on:
  push:
    branches: [ main ]
  workflow_dispatch:

jobs:
  test:
    runs-on: ubuntu-latest
    env:
      API_BASE: ${{ secrets.STAGING_API_BASE }}
      API_TOKEN: ${{ secrets.STAGING_API_TOKEN }}
    steps:
      - uses: actions/checkout@v4
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'
      - name: Install deps
        run: |
          python -m pip install --upgrade pip
          pip install pytest requests
      - name: Generate seed data
        run: python generate_seeds.py
      - name: Run tests
        run: pytest -q
```

### How to run locally and interpret results
1. **Generate CSVs**: `python generate_seeds.py`  
2. **Set env**: `export API_BASE=https://staging.api.example.com` and `export API_TOKEN=<token>`  
3. **Run tests**: `pytest -q`  
4. **CI outcome**: pipeline fails if coverage < 95% or false positive rate ≥ 2%. Test logs include failing discovery IDs and API responses for debugging.
