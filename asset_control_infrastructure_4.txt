### OpenAPI Contract (core endpoints) — YAML skeleton

Below is a **complete, extendable OpenAPI 3.1 skeleton** for the core services. Use it to generate server stubs and client SDKs.

```yaml
openapi: 3.1.0
info:
  title: Asset Control Engine API
  version: "0.1.0"
  description: Core APIs for assets, discovery, templates, checklists, workflows, and audit.

servers:
  - url: https://api.example.com
    description: Production

components:
  securitySchemes:
    oauth2:
      type: oauth2
      flows:
        clientCredentials:
          tokenUrl: https://auth.example.com/oauth2/token
          scopes:
            assets:read: Read assets
            assets:write: Write assets
            checklists:run: Run checklists
            audit:read: Read audit logs
            admin:all: Administrative actions

  schemas:
    UserRef:
      type: object
      properties:
        id:
          type: string
        name:
          type: string

    Asset:
      type: object
      required: [id, assetTag, lifecycleState]
      properties:
        id:
          type: string
        assetTag:
          type: string
        serialNumber:
          type: string
        type:
          type: string
        hostname:
          type: string
        macAddresses:
          type: array
          items:
            type: string
        ipAddresses:
          type: array
          items:
            type: string
        owner:
          $ref: '#/components/schemas/UserRef'
        lifecycleState:
          type: string
          enum: [active, quarantine, retired]
        lastSeen:
          type: string
          format: date-time
        metadata:
          type: object
          additionalProperties: true

    DiscoveryRecord:
      type: object
      properties:
        id:
          type: string
        discoverySource:
          type: string
        rawPayload:
          type: object

    ChecklistTemplate:
      type: object
      properties:
        templateId:
          type: string
        name:
          type: string
        version:
          type: integer
        steps:
          type: array
          items:
            $ref: '#/components/schemas/Step'

    Step:
      type: object
      properties:
        id:
          type: string
        type:
          type: string
          enum: [task, approval, conditional, timer, evidence]
        title:
          type: string
        assigneeRole:
          type: string
        required:
          type: boolean
        properties:
          type: object
          additionalProperties: true

    ChecklistInstance:
      type: object
      properties:
        id:
          type: string
        templateId:
          type: string
        assetId:
          type: string
        status:
          type: string
          enum: [draft, active, completed, blocked]

    AuditEntry:
      type: object
      properties:
        auditId:
          type: string
        timestamp:
          type: string
          format: date-time
        actorId:
          type: string
        actionType:
          type: string
        resourceId:
          type: string
        payloadHash:
          type: string
        prevHash:
          type: string

security:
  - oauth2: [assets:read]

paths:
  /api/assets:
    get:
      summary: List assets
      security:
        - oauth2: [assets:read]
      parameters:
        - name: filter
          in: query
          schema:
            type: string
      responses:
        '200':
          description: Asset list
          content:
            application/json:
              schema:
                type: object
                properties:
                  items:
                    type: array
                    items:
                      $ref: '#/components/schemas/Asset'
                  total:
                    type: integer

    post:
      summary: Create asset
      security:
        - oauth2: [assets:write]
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/Asset'
      responses:
        '201':
          description: Created

  /api/assets/{id}:
    get:
      summary: Get asset by id
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
      responses:
        '200':
          description: Asset
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Asset'

    patch:
      summary: Update asset
      security:
        - oauth2: [assets:write]
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              additionalProperties: true
      responses:
        '200':
          description: Updated

  /api/discovery/scan:
    post:
      summary: Trigger discovery scan
      security:
        - oauth2: [assets:write]
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                scope:
                  type: string
                connector:
                  type: string
      responses:
        '202':
          description: Scan accepted

  /api/discovery/unmatched:
    get:
      summary: List unmatched discovery records
      security:
        - oauth2: [assets:read]
      responses:
        '200':
          description: Unmatched list
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/DiscoveryRecord'

  /api/templates:
    post:
      summary: Create checklist template
      security:
        - oauth2: [checklists:run]
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ChecklistTemplate'
      responses:
        '201':
          description: Template created

    get:
      summary: List templates
      security:
        - oauth2: [checklists:run]
      responses:
        '200':
          description: Template list
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/ChecklistTemplate'

  /api/checklists:
    post:
      summary: Instantiate checklist for an asset
      security:
        - oauth2: [checklists:run]
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              properties:
                templateId:
                  type: string
                assetId:
                  type: string
      responses:
        '201':
          description: Checklist instance created
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ChecklistInstance'

  /api/checklists/{id}/steps/{stepId}/complete:
    post:
      summary: Complete a checklist step
      security:
        - oauth2: [checklists:run]
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
        - name: stepId
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                evidenceHash:
                  type: string
                notes:
                  type: string
      responses:
        '200':
          description: Step completed

  /api/workflows/trigger:
    post:
      summary: Trigger a workflow
      security:
        - oauth2: [admin:all]
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                workflowId:
                  type: string
                payload:
                  type: object
      responses:
        '202':
          description: Workflow started

  /api/audit:
    get:
      summary: Query audit entries
      security:
        - oauth2: [audit:read]
      parameters:
        - name: resourceId
          in: query
          schema:
            type: string
      responses:
        '200':
          description: Audit entries
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/AuditEntry'
```

**Notes**
- Use this YAML as the canonical contract; generate server stubs and client SDKs.
- Enforce scopes and require an additional `mfa` scope for endpoints that perform high‑risk actions (approve redeploy, vendor change).
- Add rate limits and request size limits to protect connectors.
